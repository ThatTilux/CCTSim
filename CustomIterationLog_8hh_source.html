40 // State for collecting one set of iteration data.
41 enum class DataState {

43 WAIT_FOR_ERCF, // expecting first msg(incr=0): ercf value
44 WAIT_FOR_LCF, // expecting second msg(incr=0): lcf value

59 // Helper function to trim whitespace.
60 std::string trim(const std::string &s) {
61 size_t start = s.find_first_not_of(" \t\r\n");
62 size_t end = s.find_last_not_of(" \t\r\n");
63 if (start == std::string::npos)
64 return "";
65 return s.substr(start, end - start + 1);
66 }

68 // Helper function to remove ANSI escape sequences.
69 std::string removeAnsi(const std::string &s) {
70 std::string result;
71 size_t i = 0;
72 while (i < s.size()) {
73 if (s[i] == '\033') { // ANSI escape sequence starts with ESC
74 // Skip until we hit 'm'
75 while (i < s.size() && s[i] != 'm')
76 ++i;
77 if (i < s.size())
78 ++i; // skip the 'm'
79 } else {
80 result.push_back(s[i]);
81 ++i;
82 }
83 }
84 return result;
85 }

111 void msg(const char* fmt, ...) override {
112 const int BUFFER_SIZE = 1024;
113 char buffer[BUFFER_SIZE];
114
115 va_list args;
116 va_start(args, fmt);
117 vsnprintf(buffer, BUFFER_SIZE, fmt, args);
118 va_end(args);
119
120 // Debug output.
121 //std::fprintf(stderr, "DEBUG: msg: %s\n", buffer);
122
123 std::lock_guard<std::mutex> lock(iter_mutex_);
124 std::string msgStr(buffer);
125 // Remove ANSI escape sequences.
126 std::string cleanStr = removeAnsi(msgStr);
127 std::string trimmed = trim(cleanStr);
128
129 // If we're in the start state, expect iter and fval.
130 if (current_state_ == DataState::WAIT_FOR_ITER_FVAL) {
131 if (!trimmed.empty()) {
132 std::istringstream iss(trimmed);
133 double iter_d, fval;
134 if (iss >> iter_d >> fval) {
135 temp_iter_ = static_cast<int>(iter_d);
136 temp_fval_ = fval;
137 current_state_ = DataState::WAIT_FOR_ERCF;
138 }
139 }
140 }
141 // In either WAIT_FOR_OPTIONAL_CCF or WAIT_FOR_SET_END state,
142 // a blank message signals the end of the set.
143 else if ((current_state_ == DataState::WAIT_FOR_OPTIONAL_CCF ||
144 current_state_ == DataState::WAIT_FOR_SET_END) &&
145 trimmed.empty()) {
146 // Commit the collected values.
147 last_values_.iter = temp_iter_;
148 last_values_.fval = temp_fval_;
149 last_values_.ercf = temp_ercf_;
150 last_values_.lcf = temp_lcf_;
151 last_values_.ccf = (current_state_ == DataState::WAIT_FOR_SET_END) ? temp_ccf_ : 0.0;
152 last_values_.has_ccf = (current_state_ == DataState::WAIT_FOR_SET_END);
153 // Reset state for next set.
154 current_state_ = DataState::WAIT_FOR_ITER_FVAL;
155 }
156 // Otherwise, ignore messages.
157 }

169 void msg(const int incr, const char* fmt, ...) override {
170 const int BUFFER_SIZE = 1024;
171 char buffer[BUFFER_SIZE];
172
173 va_list args;
174 va_start(args, fmt);
175 vsnprintf(buffer, BUFFER_SIZE, fmt, args);
176 va_end(args);
177
178 // Debug output.
179 //std::fprintf(stderr, "DEBUG: msg(incr=%d): %s\n", incr, buffer);
180
181 // If incr is non-zero, update indentation and exit.
182 if (incr != 0) {
183 num_indent_ += incr; // update parent's indentation if needed.
184 return;
185 }
186
187 std::lock_guard<std::mutex> lock(iter_mutex_);
188 std::string msgStr(buffer);
189 std::string cleanStr = removeAnsi(msgStr);
190 std::string trimmed = trim(cleanStr);
191 if (trimmed.empty()) return; // ignore blank messages here
192
193 std::istringstream iss(trimmed);
194 double value;
195 if (!(iss >> value)) return; // if parsing fails, ignore
196
197 // Depending on the current state, assign the value.
198 if (current_state_ == DataState::WAIT_FOR_ERCF) {
199 temp_ercf_ = value;
200 current_state_ = DataState::WAIT_FOR_LCF;
201 } else if (current_state_ == DataState::WAIT_FOR_LCF) {
202 temp_lcf_ = value;
203 current_state_ = DataState::WAIT_FOR_OPTIONAL_CCF;
204 } else if (current_state_ == DataState::WAIT_FOR_OPTIONAL_CCF) {
205 temp_ccf_ = value;
206 temp_ccf_provided_ = true;
207 current_state_ = DataState::WAIT_FOR_SET_END;
208 }
209 // Ignore messages if not expected.
210 }
